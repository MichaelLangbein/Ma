\newcommand{\induction}[5]{
    \subprf{By induction.  Let $P(i)$ be #1 }{$P(1)$ and $P(i) \to P(i+1)$ hold true.}{
        \subprf{Base case: #2 }{$P(1)$ holds true.}{
            #3
        }
        \\
        \subprf{Induction step: #4 }{$P(i) \to P(i+1)$ holds true.}{
            #5
        }
    }
}
